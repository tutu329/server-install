# -------------------------------------最简推理脚本---------------------------------------------
cat > art_infer_smoke.py <<'PY'
import os
import asyncio
import multiprocessing as mp

import art
from art.local.backend import LocalBackend
from art.dev import InternalModelConfig

MODEL = os.environ.get("MODEL", "/home/tutu/models/Qwen3-0.6B")


async def main():
    print("开始配置 backend")

    backend = LocalBackend(
        path="./.art-qwen3-06b-smoke",
        # 先用 in_process=True，减少多进程问题。
        # 等这个通了，再改回 False 测正式子进程模式。
        in_process=False,
    )

    model = art.TrainableModel(
        name="qwen3-06b-infer-smoke",
        project="art-smoke",
        base_model=MODEL,
        lora_config={
            "rank": 8,
            "alpha": 16,
            "dropout": 0.0,
            "target_modules": [
                "q_proj",
                "k_proj",
                "v_proj",
                "o_proj",
                "gate_proj",
                "up_proj",
                "down_proj",
            ],
        },
        _internal_config=InternalModelConfig(
            allow_unvalidated_arch=True,
            init_args={
                "dtype": "float16",
                # 0.8B 在 22G 2080Ti 上不需要 4bit，先关闭 bitsandbytes。
                "load_in_4bit": False,
                "max_seq_length": 2048,
                "trust_remote_code": True,
            },
            engine_args={
                "dtype": "float16",
                "max_model_len": 2048,
                "gpu_memory_utilization": 0.70,
                "trust_remote_code": True,
                "enforce_eager": True,
            },
            trainer_args={
                "bf16": False,
                "fp16": True,

                "per_device_train_batch_size": 1,
                "gradient_accumulation_steps": 1,

                # 关键修复：generation_batch_size 必须能被 num_generations 整除
                "num_generations": 2,
                "generation_batch_size": 2,

                "max_completion_length": 64,
            },
        ),
    )

    print("开始 register")
    await model.register(backend)
    print("register 完成")

    client = model.openai_client()

    print("开始推理")
    resp = await client.chat.completions.create(
        model=model.get_inference_name(),
        messages=[
            {"role": "system", "content": "你是一个简洁的中文助手。"},
            {"role": "user", "content": "只回答一句话：ART推理是否正常？"},
        ],
        max_tokens=64,
        temperature=0.0,
        timeout=300,
    )

    print("BASE_MODEL =", MODEL)
    print("INFERENCE_NAME =", model.get_inference_name())
    print("ANSWER =", resp.choices[0].message.content)

    await backend.close()


if __name__ == "__main__":
    mp.freeze_support()
    asyncio.run(main())
PY

解决手段：打好下列补丁（所在文件夹，保持上述测试时文件夹不变），然后继续执行测试，即可看到vllm log中的health成功结果("GET /health HTTP/1.1" 200 OK)，python art_infer_smoke.py也能正常运行出结果：
# -------------------------------解决art调用vllm的health报错的补丁脚本---------------------------------------------
source ~/venvs/art-vllm-runtime/bin/activate

python - <<'PY'
from pathlib import Path
import re
import py_compile
import prometheus_fastapi_instrumentator.routing as r

p = Path(r.__file__)
src = p.read_text()

backup = p.with_suffix(".py.bak_func")
if not backup.exists():
    backup.write_text(src)
    print("backup:", backup)

new_func = '''def _get_route_name(
    scope: Scope, routes: List[Route], route_name: Optional[str] = None
) -> Optional[str]:
    """Gets route name for given scope taking mounts into account."""

    for route in routes:
        # vLLM/FastAPI newer router internals may include _IncludedRouter objects.
        # prometheus-fastapi-instrumentator expects Starlette Route-like objects.
        # Skip anything that does not expose both matches() and path.
        if not hasattr(route, "matches"):
            continue

        path = getattr(route, "path", None)
        if path is None:
            continue

        match, child_scope = route.matches(scope)
        if match == Match.FULL:
            route_name = path
            child_scope = {**scope, **child_scope}
            if isinstance(route, Mount) and getattr(route, "routes", None):
                child_route_name = _get_route_name(child_scope, route.routes, route_name)
                if child_route_name is None:
                    route_name = None
                else:
                    route_name += child_route_name
            return route_name
        elif match == Match.PARTIAL and route_name is None:
            route_name = path

    return None


'''

pattern = r'def _get_route_name\([\s\S]*?\n\n\ndef get_route_name'
replacement = new_func + "def get_route_name"

src2, n = re.subn(pattern, replacement, src, count=1)

if n != 1:
    raise SystemExit(f"替换失败，匹配次数={n}")

p.write_text(src2)
py_compile.compile(str(p), doraise=True)

print("patched ok:", p)
PY

三、ART的最简rl训练测试
1、构建训练脚本：
# -------------------------------------rl训练脚本---------------------------------------------
cat > art_rl_smoke.py <<'PY'
import os
import asyncio
import multiprocessing as mp

import art
from art.local import LocalBackend
from art.dev import InternalModelConfig

MODEL = os.environ.get("MODEL", "/home/tutu/models/Qwen3-0.6B")

TARGET_MODULES = [
    "q_proj",
    "k_proj",
    "v_proj",
    "o_proj",
    "gate_proj",
    "up_proj",
    "down_proj",
]


async def rollout(model: art.Model, prompt: str, rollout_idx: int) -> art.Trajectory:
    client = model.openai_client()

    trajectory = art.Trajectory(
        messages_and_choices=[
            {"role": "system", "content": "你是一个简洁助手。"},
            {"role": "user", "content": prompt},
        ]
    )

    completion = await client.chat.completions.create(
        model=model.get_inference_name(),
        messages=trajectory.messages(),
        max_tokens=32,
        temperature=1.0,
        logprobs=True,
        timeout=300,
    )

    choice = completion.choices[0]
    trajectory.messages_and_choices.append(choice)

    # 冒烟测试用的人为奖励：每个 prompt 下两个 completion 分别给 +1 / -1，
    # 保证同一个 GRPO group 内有奖励差异。
    trajectory.reward = 1.0 if rollout_idx % 2 == 0 else -1.0
    trajectory.metrics["smoke_reward"] = trajectory.reward

    return trajectory


async def main():
    backend = LocalBackend(
        path="./.art-qwen3-06b-rl",
        in_process=False,
    )

    try:
        model = art.TrainableModel(
            name="qwen3-06b-rl-smoke",
            project="art-smoke",
            base_model=MODEL,
            lora_config={
                "rank": 8,
                "alpha": 16,
                "dropout": 0.0,
                "target_modules": TARGET_MODULES,
            },
            _internal_config=InternalModelConfig(
                allow_unvalidated_arch=True,
                init_args={
                    "dtype": "float16",
                    "load_in_4bit": False,
                    "max_seq_length": 2048,
                    "trust_remote_code": True,
                },
                engine_args={
                    "dtype": "float16",
                    "max_model_len": 2048,
                    "gpu_memory_utilization": 0.70,
                    "trust_remote_code": True,
                    "enforce_eager": True,
                },
                trainer_args={
                    "bf16": False,
                    "fp16": True,

                    # 关键修复：
                    # 你的日志显示训练第一步死在 bitsandbytes 8-bit optimizer，
                    # 缺 libnvJitLink.so.13，所以 smoke 阶段先强制用 PyTorch AdamW。
                    "optim": "adamw_torch",

                    "per_device_train_batch_size": 1,
                    "gradient_accumulation_steps": 1,

                    # generation_batch_size 必须能被 num_generations 整除。
                    "num_generations": 2,
                    "generation_batch_size": 2,

                    "max_completion_length": 64,
                },
            ),
        )

        await model.register(backend)

        print("BASE_MODEL =", MODEL)
        print("INFERENCE_NAME =", model.get_inference_name())
        print("STEP before =", await model.get_step())

        scenarios = [
            "请回答：2+2=?",
            "请回答：北京是哪个国家的首都？",
        ]

        train_groups = await art.gather_trajectory_groups(
            [
                art.TrajectoryGroup(
                    rollout(model, scenario, rollout_idx)
                    for rollout_idx in range(2)
                )
                for scenario in scenarios
            ],
            pbar_desc="gather",
        )

        train_groups = [g for g in train_groups if len(g) >= 2]

        print("num train groups =", len(train_groups))
        print("group sizes =", [len(g) for g in train_groups])

        result = await backend.train(
            model,
            train_groups,
            learning_rate=1e-6,
            verbose=True,
        )

        await model.log(
            train_groups,
            metrics=result.metrics,
            step=result.step,
            split="train",
        )

        print("STEP after =", await model.get_step())
        print("checkpoint =", result.checkpoint_path)
        print("metrics =", result.metrics)

    finally:
        await backend.close()


if __name__ == "__main__":
    mp.freeze_support()
    asyncio.run(main())

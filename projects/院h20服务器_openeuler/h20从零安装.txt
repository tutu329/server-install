[Unit]
Description=frpc Service
After=network.target
Wants=network.target

[Service]
Type=simple
User=root
Restart=always
RestartSec=5s
ExecStart=/root/download/frp_0.51.3_linux_amd64/frpc -c /root/download/frp_0.51.3_linux_amd64/frpc.ini
ExecReload=/root/download/frp_0.51.3_linux_amd64/frpc -c /root/download/frp_0.51.3_linux_amd64/frpc.ini

[Install]
WantedBy=multi-user.target


# frpc.ini
[common]
server_addr = powerai.cc
server_port = 7880

[ssh2 26000]
type = tcp
local_ip = 127.0.0.1
local_port = 22
remote_port = 26000

[ftp2 28021]
type = tcp
local_ip = 127.0.0.1
local_port = 21
remote_port = 28021

# if you want to expose multiple ports, add 'range:' prefix to the section name
# frpc will generate multiple proxies such as 'tcp_port_6010', 'tcp_port_6011' and so on.
[range:ftp_pasv1]
type = tcp
local_ip = 127.0.0.1
local_port = 28012-28020
remote_port = 28012-28020
use_encryption = false
use_compression = false

[range:sd2]
type = tcp
local_ip = 127.0.0.1

remote_port = 25100-25120

[range:llm2]

[range:ui2]
